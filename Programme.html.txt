function show(id,btn){
  document.querySelectorAll('section').forEach(s=>s.classList.remove('active'));
  document.querySelectorAll('nav button').forEach(b=>b.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  if(btn)btn.classList.add('active');
  window.scrollTo({top:0,behavior:'smooth'});
}
let logs=[];
try{logs=JSON.parse(sessionStorage.getItem('pclogs')||'[]')}catch(e){}
function saveDay(){
  const p=document.getElementById('tp').value||'—';
  const t=document.getElementById('tt').value||'—';
  const e=document.getElementById('te').value||'—';
  const f=document.getElementById('tf').value||'—';
  const r=document.getElementById('tr2').value||'—';
  const d=new Date().toLocaleDateString('fr-FR',{weekday:'short',day:'numeric',month:'short'});
  logs.unshift({date:d,p,t,e,f,r});
  try{sessionStorage.setItem('pclogs',JSON.stringify(logs))}catch(e){}
  renderLogs();
  ['tp','tt','te','tf','tr2'].forEach(id=>{document.getElementById(id).value=''});
  const btn=document.querySelector('.sbtn');
  btn.textContent='✅ ENREGISTRÉ !';
  setTimeout(()=>btn.textContent='ENREGISTRER MA JOURNÉE',2000);
}

  const c=document.getElementById('logcard');
  const l=document.getElementById('loglist');
  if(!logs.length){c.style.display='none';return}
  c.style.display='block';
  l.innerHTML=logs.map(e=>`<div class="le"><span class="ld">${e.date}</span><span>💪${e.p} &nbsp;🏃${e.t} tours &nbsp;😊${e.r}/10</span></div>`).join('');
}
